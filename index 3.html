function field(value) {
    const ocr = "%ocr%";
    let match = null;

    const regexArr = [
        /Faktura\/betalingsdato\s([0-9]{1,2})[.,\s]+([A-Za-z]+)\s([0-9]{2,4})\s/
    ];

    const monthMap = {
    januar: "01",
    februar: "02",
    marts: "03",
    april: "04",
    maj: "05",
    juni: "06",
    juli: "07",
    august: "08",
    september: "09",
    oktober: "10",
    november: "11",
    december: "12"
};

    for (var i = 0; i < regexArr.length; i++) {
        match = ocr.match(regexArr[i]);
        if (match) break;
    }

    if (!match) return "";

    var day = match[1];
    if (day.length === 1) day = "0" + day;

    var monthText = match[2].toLowerCase();
    var month = monthMap[monthText] || "";

    var year = match[3];
    if (year.length === 2) {
        year = "20" + year;
    }

    return year + "-" + month + "-" + day;
}
